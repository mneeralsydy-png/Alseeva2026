        async function saveAttendance() {
            const date = document.getElementById('attendanceDate').value;
            const halakaId = document.getElementById('attendanceHalaka').value;

            if (!date || !halakaId) {
                showAlert('attendanceAlert', '❌ الرجاء اختيار التاريخ والحلقة', 'error');
                return;
            }

            try {
                const snapshot = await db.collection('students').get();
                let count = 0;

                snapshot.forEach(doc => {
                    const statusEl = document.getElementById(`status_${doc.id}`);
                    const notesEl = document.getElementById(`notes_${doc.id}`);

                    if (statusEl) {
                        db.collection('attendance').add({
                            studentId: doc.id,
                            studentName: doc.data().name,
                            date: date,
                            halakaId: halakaId,
                            status: statusEl.value,
                            notes: notesEl?.value || '',
                            createdAt: firebase.firestore.FieldValue.serverTimestamp()
                        });
                        count++;
                    }
                });

                showAlert('attendanceAlert', `✅ تم تسجيل حضور ${count} طالب بنجاح! 🎉`, 'success');
            } catch (err) {
                showAlert('attendanceAlert', '❌ خطأ: ' + err.message, 'error');
            }
        }

        function uploadImages(event) {
            event.preventDefault();

            const album = document.getElementById('albumSelect').value;
            const files = document.getElementById('imageFiles').files;
            const description = document.getElementById('imageDescription').value;
            const btn = document.getElementById('uploadBtn');

            if (!files.length) {
                showAlert('mediaAlert', '❌ الرجاء اختيار صور', 'error');
                return false;
            }

            btn.disabled = true;
            btn.innerHTML = '<span class="loading-spinner"></span> جاري الرفع...';

            let uploaded = 0;

            Array.from(files).forEach(file => {
                const storageRef = storage.ref(`images/${album}/${Date.now()}_${file.name}`);
                
                storageRef.put(file).then(snapshot => {
                    return snapshot.ref.getDownloadURL();
                }).then(url => {
                    db.collection('media').add({
                        album: album,
                        imageUrl: url,
                        imageName: file.name,
                        description: description,
                        uploadedAt: firebase.firestore.FieldValue.serverTimestamp()
                    });
                    uploaded++;

                    if (uploaded === files.length) {
                        showAlert('mediaAlert', `✅ تم رفع ${files.length} صور بنجاح! 🎉`, 'success');
                        btn.disabled = false;
                        btn.innerHTML = '☁️ رفع الصور إلى Firebase';
                        event.target.reset();
                    }
                }).catch(err => {
                    console.error('Upload error:', err);
                    showAlert('mediaAlert', '❌ خطأ في رفع: ' + err.message, 'error');
                    btn.disabled = false;
                    btn.innerHTML = '☁️ رفع الصور إلى Firebase';
                });
            });

            return false;
        }

        function deleteDoc(collection, id, name) {
            if (confirm(`هل تريد حذف "${name}"؟`)) {
                db.collection(collection).doc(id).delete()
                    .then(() => alert('✅تم الحذف!'))
                    .catch(err => alert('❌خطأ: ' + err.message));
            }
        }

        // ============================================
        // UTILITY FUNCTIONS
        // ============================================
        
        function getBadgeColor(level) {
            switch(level) {
                case 'متقدم': return 'badge-success';
                case 'متوسط': return 'badge-warning';
                default: return 'badge-info';
            }
        }

        function getTimeAgo(seconds) {
            if (!seconds) return 'الآن';
            
            const now = Date.now() / 1000;
            const diff = now - seconds;

            if (diff < 60) return 'الآن';
            if (diff < 3600) return `منذ ${Math.floor(diff/60)} دقيقة`;
            if (diff < 86400) return `منذ ${Math.floor(diff/3600)} ساعة`;
            if (diff < 604800) return `منذ ${Math.floor(diff/86400)} يوم`;

            return new Date(seconds * 1000).toLocaleDateString('ar-SA');
        }
